## 첫번째 시도 

```cs
    public class Solution
    {
        public string solution(string X, string Y)
        {
            string answer = "";
            List<int> list = new List<int>();
            int[] XList = new int[X.Length];
            int[] YList = new int[Y.Length];

            for (int i = 0; i < X.Length; i++)
            {
                int num =  X[i] - '0';
                XList[i] = num;
            }
            for (int i = 0; i < Y.Length; i++)
            {
                int num = Y[i] - '0';
                YList[i] = num;
            }

            XList.ToList().Sort();
            YList.ToList().Sort();

            for (int i = 0; i < X.Length; i++)
            {

                int index = YList.ToList().IndexOf(XList[i]);

                if (index > -1)     // 값이 있다는 뜻
                {
                    list.Add(YList[index]);

                    YList[index] = int.MinValue;
                }
            }

            list.Sort();
            list.Reverse();

            while (list[0] == 0)
            {
                if (list.Count == 1) break;
                list.Remove(0);
            }

            if (list.Count <= 0)
            {
                answer = "-1";
                return answer;
            }

            for (int i = 0; i < list.Count; i++)
            {
                answer += list[i];
            }

            return answer;
        }
    }
```

## 두번째 시도 

```cs
 public string solution(string X, string Y)
        {
            string answer = "";

            int[] Xnums = new int[9];
            int[] Ynums = new int[9];

            foreach (var item in X)
            {
                int index = item - '0';
                Xnums[index]++;
                
            }
            foreach (var item in Y)
            {
                int index = item - '0';
                Ynums[index]++;
            }

            for (int i = 0; i < 9; i++)
            {
                if (Xnums[i] == 0 || Ynums[i] == 0) continue;

                for (int j = 0; j < MathF.Min(Xnums[i], Ynums[i]); j++)
                {
                    answer += i.ToString();
                }
                
            }

            answer = new string(answer.Reverse().ToArray());
            
            if(answer.ToCharArray().Length <= 0)
            {
                answer = "-1";
                return answer;
            }

            while (answer[0] == '0')
            {
                if (answer.ToCharArray().Length == 1) break;
                answer = answer.Substring(1);
            }

            return answer;
        }
```
