## 첫번째 풀이

```cs
public string solution(string s, string skip, int index)
        {
            string answer = "";

            foreach (char item in s)
            {
                char tempS = item ;

                for (int i = 0; i < index; i++)
                {
                    
                    tempS = Check(tempS, skip);
                }

                answer += tempS;
            }

            return answer;
        }

        char Check(char item, string skip)
        {
            
            char tempS = (char)(item + (char)1); 

            if(skip.Contains(tempS))
            {
                tempS = Check(tempS, skip);
            }

            if(tempS == '{')
            {
                tempS = 'a';
            }

            return tempS;

        }
```

몇몇 테스트 케이스가 실패 했다. 질문하기 질문들을 보니 skip 정렬해서 돌릴 경우 a와 z가 둘다 skip에 있을 때 z -> a로 넘어가고 b로 넘어가지 않을 수 있다는 글을 발견했다.

## 두번째 풀이
```cs
public string solution(string s, string skip, int index)
        {
            string answer = "";

            foreach (char item in s)
            {
                char tempS = item ;

                for (int i = 0; i < index; i++)
                {
                    
                    tempS = Check(tempS, skip);

                    if (skip.Contains(tempS))
                    {
                        tempS = Check(tempS, skip);
                    }
                }

                answer += tempS;
            }

            return answer;
        }

        char Check(char item, string skip)
        {
            
            char tempS = (char)(item + (char)1); 

            if(skip.Contains(tempS))
            {
                tempS = Check(tempS, skip);
            }

            if (tempS == '{')
            {
                tempS = 'a';
            }


            return tempS;

        }
```
